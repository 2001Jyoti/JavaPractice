//Relational Operator
public class RelationalOperator {
 public static void main(String[] args) {
   int x=3;
   int y=4;
   System.out.println("x>y:"+(x>y));
   System.out.println("x<y:"+(x<y));
   System.out.println("x>=y:"+(x>=y));
   System.out.println("x<=y:"+(x<=y));
   System.out.println("x==y:"+(x==y));
   System.out.println("x!=y:"+(x!=y));
  }
}
